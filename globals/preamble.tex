% format section headings
\usepackage{titlesec}
\titleformat{\section}
  {\Large\bfseries\color{schoolcolorprimary}}
  {\thesection.}
  {0.5em}
  {}
  [{\color{schoolcolorsecondary}\titlerule[1pt]}]

\titlespacing*{\section}
  {0pt}  % left
  {14pt} % before
  {6pt}  % after

% format subsection headings
\titleformat{\subsection}
  {\large\bfseries}
  {\thesubsection.}
  {0.5em}
  {}

\titlespacing*{\subsection}
  {0pt}  % left
  {9pt}  % before
  {3pt}  % after

\titleformat{\subsubsection}
  {\large\itshape}
  {\thesubsubsection.}
  {0.5em}
  {}

\titlespacing*{\subsubsection}
  {0pt}  % left
  {3pt}  % before
  {3pt}  % after
